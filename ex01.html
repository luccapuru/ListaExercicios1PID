Neste primeiro exercício, foi pedido que fosse carregado uma imagem digital em níveis de cinza. A imagem escolhida foi uma presente no Oxford Buildings Dataset, como pode ser observado a seguir.

import cv2 
import numpy as np
from matplotlib import pyplot as plt

#carregando a imagem
img = cv2.imread("images//all_souls_000003.jpg", cv2.IMREAD_GRAYSCALE)
#mostrando imagem
plt.imshow(img, cmap = "gray")
plt.xticks([]), plt.yticks([])
plt.title("Imagem Original")
plt.show()
